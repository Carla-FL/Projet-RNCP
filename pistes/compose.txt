version: "3.8"

services:

  mongodb:
    image: mongo:latest
    container_name: mongodb
    ports:
      - "27017:27017"
    environment:
      MONGO_INITDB_ROOT_USERNAME: admin
      MONGO_INITDB_ROOT_PASSWORD: pass123
    volumes:
      - mongo_data:/data/db

  extractor:
    build: ./extractor
    container_name: extractor
    depends_on:
      - mongodb
    environment:
      - MONGO_URI=mongodb://admin:pass123@mongodb:27017
    command: ["python", "main.py"]

  sentiment:
    build: ./sentiment

    depends_on:
      - mongodb
    environment:
      - MONGO_URI=mongodb://admin:pass123@mongodb:27017
    command: ["python", "main.py"]

  topics:
    build: ./topics
    container_name: topics
    depends_on:

      - MONGO_URI=mongodb://admin:pass123@mongodb:27017
    command: ["python", "main.py"]
